---
#tasks file

  - block: #Linux
      - name: install
        yum: name: "{{services}}" state=latest
      - name: start
        service: name="{{services}}" state=started enable=yes
    when: ansible-os_family == "RedHat"


  - block: #Windows
      - name: install
        win_package: name="{{services}}" state=started enable=yes
      - name: start
        service: name="{{services}}" state=started enable=yes
    when: ansible-os_family == "Windows"
